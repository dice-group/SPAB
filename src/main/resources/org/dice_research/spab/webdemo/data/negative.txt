SELECT DISTINCT  ?resource ?uri ?wtitle ?comment ?image
WHERE
  { { { ?resource foaf:page <http://en.wikipedia.org/wiki/Telecommunication> } }
    ?resource foaf:page ?uri .
    ?resource rdfs:label ?wtitle
    FILTER langMatches(lang(?wtitle), "en")
    OPTIONAL
      { ?resource rdfs:comment ?comment
        FILTER langMatches(lang(?comment), "en")
      }
    OPTIONAL
      { ?resource foaf:depiction ?image }
  }
